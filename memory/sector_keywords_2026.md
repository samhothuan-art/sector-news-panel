## 七、使用说明

### 在监控脚本中使用

```python
# 板块关键词库
SECTOR_KEYWORDS = {
    "AI算力": ["AI", "大模型", "算力", "光模块", "CPO", "液冷", "服务器"],
    "半导体": ["半导体", "芯片", "光刻机", "存储", "先进封装", "国产替代"],
    "机器人": ["人形机器人", "减速器", "丝杠", "传感器", "Optimus"],
    "新能源": ["新能源车", "锂电池", "光伏", "储能", "逆变器"],
    "有色": ["铜", "铝", "黄金", "涨价", "供需"],
    "消费": ["白酒", "医药", "旅游", "免税", "银发经济"],
}

# 过滤新闻
def filter_news_by_sector(news_list, sector_keywords):
    sector_news = {sector: [] for sector in sector_keywords}
    
    for news in news_list:
        title = news['title']
        for sector, keywords in sector_keywords.items():
            if any(kw in title for kw in keywords):
                sector_news[sector].append(news)
                break
    
    return sector_news
```
